def get_ai_insights(predictor, recent_data, api_key):
    """Ottiene insights AI sul sistema di trading."""
    chatbot = create_trading_chatbot(api_key)
    
    # Predizione corrente
    current_pred = predictor.predict_single(recent_data, return_confidence=True)
    
    # Spiegazione AI
    explanation = chatbot.explain_prediction(
        prediction_result=current_pred,
        market_context={"volatility": calculate_volatility(recent_data)}
    )
    
    return explanation
